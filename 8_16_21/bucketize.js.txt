function bucketize(s, n) {
  let word = s.split(" ");
  let last = word[0];
  let answer = [];
  if (s.length < n) {
    return answer;
  } 
  else {
    for (let i = 1; i < word.length; i++) {
      if (last.length + word[i].length <= n-1) {
        let temp = last + " " + word[i];
        last = temp;
      } else {
        answer.push(last);
        last = word[i];
      }

      if (i === word.length - 1) {
        answer.push(last);
      }
    }
    return answer;
  }
}
